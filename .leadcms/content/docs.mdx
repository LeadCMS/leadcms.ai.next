## Quick Start

Get up and running with LeadCMS in minutes:

<Steps>
  <Step title="Create Account">
    Sign up for a free LeadCMS account at [leadcms.io](https://leadcms.io/signup)
  </Step>
  <Step title="Install CLI">
    ```bash
    npm install -g @leadcms/cli
    ```
  </Step>
  <Step title="Create Project">
    ```bash
    leadcms init my-project
    cd my-project
    npm run dev
    ```
  </Step>
</Steps>

Your site will be running at `http://localhost:3000` with sample content ready to customize.

## Code Examples

Here are some common patterns to get you started:

### Fetching Content

```typescript
import { LeadCMS } from '@leadcms/client'

const cms = new LeadCMS({
  apiKey: process.env.LEADCMS_API_KEY,
  projectId: process.env.LEADCMS_PROJECT_ID
})

// Get a single page
const page = await cms.content.getBySlug('about-us')

// List articles with pagination
const articles = await cms.content.list({
  type: 'article',
  status: 'published',
  limit: 10,
  sort: 'publishedAt:desc'
})
```

### React Component

```tsx
import { useContent } from '@leadcms/react'

function BlogPost({ slug }: { slug: string }) {
  const { data: post, loading, error } = useContent('article', slug)

  if (loading) return <div>Loading...</div>
  if (error) return <div>Error loading post</div>

  return (
    <article>
      <h1>{post.title}</h1>
      <time>{new Date(post.publishedAt).toLocaleDateString()}</time>
      <div dangerouslySetInnerHTML={{ __html: post.content }} />
    </article>
  )
}
```

### Static Site Generation

```typescript
// Next.js static generation
export async function getStaticProps() {
  const cms = new LeadCMS({ /* config */ })
  const page = await cms.content.getBySlug('homepage')

  return {
    props: { page },
    revalidate: 3600 // Revalidate every hour
  }
}

export async function getStaticPaths() {
  const cms = new LeadCMS({ /* config */ })
  const pages = await cms.content.list({ type: 'page' })

  return {
    paths: pages.map(page => ({ params: { slug: page.slug } })),
    fallback: 'blocking'
  }
}
```
